package jp.co.misumi.replenishstock.client;

import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.mastermanagement.internal.api.SubsidiaryInternalApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.SubsidiaryInternalClient;
import jp.co.misumi.replenishstock.model.rest.client.mastermanagement.internal.GetSubsidiaryInternalResponseV1;
import lombok.RequiredArgsConstructor;

@RequiredArgsConstructor
@Named
public class DefaultSubsidiaryInternalClient implements SubsidiaryInternalClient {
  private final HttpHeaderMapCreator httpHeaderMapCreator;
  private final SubsidiaryInternalApi subsidiaryApi;

  @Override
  public GetSubsidiaryInternalResponseV1 getSubsidiaryInternalV1(
      RequestContext context, String controlCode) throws Exception {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return subsidiaryApi.getSubsidiaryInternalV1(
        context,
        ClientEndpoint.CLIENT_ENDPOINT_GET_SUBSIDIARY.getValue(),
        headerMap,
        null,
        controlCode);
  }
}
