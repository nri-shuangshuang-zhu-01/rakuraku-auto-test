package jp.co.misumi.replenishstock.client;

import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.purchaseorder.internal.api.PoApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.PoClient;
import jp.co.misumi.replenishstock.model.rest.client.purchaseorder.internal.SearchDetailAllPoInternalRequestV1;
import jp.co.misumi.replenishstock.model.rest.client.purchaseorder.internal.SearchDetailAllPoInternalResponseV1;
import lombok.RequiredArgsConstructor;
import org.springframework.web.client.RestClientException;

/**
 * 過少値算出.
 *
 * @author sho.daii
 * @since 1.0.0
 */
@RequiredArgsConstructor
@Named
public class DefaultPoClient implements PoClient {
  /**
   * 発注詳細全体検索API 発注詳細全体検索API（ID:MAPPOPO250）
   *
   * <p><b>200</b> - 発注詳細全体検索APIレスポンス
   *
   * <p><b>400</b> - エラーレスポンス
   *
   * @param context The RequestContext of this request
   * @param searchDetailAllPoInternalRequestV1 発注詳細全体検索APIリクエスト (optional)
   * @return SearchDetailAllPoInternalResponseV1
   * @throws RestClientException if an error occurs while attempting to invoke the API
   */
  private final HttpHeaderMapCreator httpHeaderMapCreator;

  private final PoApi poApi;

  @Override
  public SearchDetailAllPoInternalResponseV1 searchDetailAllPoV1(
      RequestContext context, SearchDetailAllPoInternalRequestV1 mappRequest)
      throws RestClientException {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return poApi.searchDetailAllPoV1(
        context, ClientEndpoint.CLIENT_ENDPOINT_MAPPOP_O250.getValue(), headerMap, mappRequest);
  }
}
