package jp.co.misumi.replenishstock.client;

import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.stock.internal.api.StockInfoApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.StockInfoClient;
import jp.co.misumi.replenishstock.model.rest.client.stock.internal.CheckStockInfoRequestV1;
import jp.co.misumi.replenishstock.model.rest.client.stock.internal.CheckStockInfoResponseV1;
import jp.co.misumi.replenishstock.model.rest.client.stock.internal.UpdateStockInfoRequestV1;
import jp.co.misumi.replenishstock.model.rest.client.stock.internal.UpdateStockInfoResponseV1;
import lombok.RequiredArgsConstructor;
import org.springframework.web.client.RestClientException;
/**
 * 在庫情報更新API.
 *
 * @author sho.daii
 * @since 1.0.0 @Date 2021/09/12
 */
@RequiredArgsConstructor
@Named
public class DefaultStockInfoClient implements StockInfoClient {
  private final StockInfoApi stockInfoApi;
  private final HttpHeaderMapCreator httpHeaderMapCreator;

  @Override
  public CheckStockInfoResponseV1 checkStockInfoV1(
      RequestContext context, CheckStockInfoRequestV1 checkStockInfoRequestV1)
      throws RestClientException {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return stockInfoApi.checkStockInfoV1(
        context,
        ClientEndpoint.CLIENT_ENDPOINT_CHECK_STOCK_INFO.getValue(),
        headerMap,
        "1",
        checkStockInfoRequestV1);
  }

  @Override
  public UpdateStockInfoResponseV1 updateStockInfoV1(
      RequestContext context, UpdateStockInfoRequestV1 updateStockInfoRequestV1)
      throws RestClientException {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return stockInfoApi.updateStockInfoV1(
        context,
        ClientEndpoint.CLIENT_ENDPOINT_UPDATE_STOCK_INFO.getValue(),
        headerMap,
        "1",
        updateStockInfoRequestV1);
  }
}
