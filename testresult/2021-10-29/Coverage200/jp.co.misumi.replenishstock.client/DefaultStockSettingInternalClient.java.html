package jp.co.misumi.replenishstock.client;

import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.stock.internal.api.StockSettingInternalApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.StockSettingInternalClient;
import jp.co.misumi.replenishstock.model.rest.client.stock.internal.GetStockSettingInternalResponseV1;
import lombok.RequiredArgsConstructor;
import org.springframework.web.client.RestClientException;

/**
 * DefaultStockSettingInternalClient.
 *
 * <p>coding by LKGæ¬§
 *
 * @since 1.0.0
 */
@RequiredArgsConstructor
@Named("defaultStockSettingInternalClient")
public class DefaultStockSettingInternalClient implements StockSettingInternalClient {
  private final HttpHeaderMapCreator httpHeaderMapCreator;
  private final StockSettingInternalApi stockSettingInternalApi;

  @Override
  public GetStockSettingInternalResponseV1 getStockSettingInternalV1(
      RequestContext context, String ginnerCode, String plantCode) throws RestClientException {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return stockSettingInternalApi.getStockSettingInternalV1(
        context,
        ClientEndpoint.CLIENT_ENDPOINT_GET_STOCK_SETTING_V1_INTERNAL.getValue(),
        headerMap,
        ginnerCode,
        plantCode);
  }
}
