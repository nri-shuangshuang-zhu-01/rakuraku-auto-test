package jp.co.misumi.replenishstock.client;

import java.io.File;
import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.fileuploadcontainer.yaml.container.api.FileApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.GeneralUploadClient;
import jp.co.misumi.replenishstock.model.rest.client.fileuploadcontainer.yaml.container.FileUploadResponseV1;
import lombok.RequiredArgsConstructor;

@RequiredArgsConstructor
@Named
public class DefaultGeneralUploadClient implements GeneralUploadClient {
  private final FileApi fileApi;
  private final HttpHeaderMapCreator httpHeaderMapCreator;

  /**
   * 汎用ファイルアップロード.
   *
   * @param context The RequestContext of this request
   * @param file The file of this request
   * @return FileUploadResponseV1
   */
  @Override
  public FileUploadResponseV1 uploadFileV1(RequestContext context, File file) {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    return fileApi.uploadFileV1(
        context, ClientEndpoint.CLIENT_UPLOAD_FILE_V1.getValue(), headerMap,  file);
  }
}
