package jp.co.misumi.replenishstock.store.repository.mybatis;

import java.util.Optional;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.replenishstock.domain.entity.WeeklyReorderPointPoolDataMonthlyEntity;
import jp.co.misumi.replenishstock.domain.repository.WeeklyReorderPointPoolMonthlyRepository;
import lombok.NonNull;
import lombok.RequiredArgsConstructor;
import lombok.experimental.Delegate;

@RequiredArgsConstructor
public class MybatisWeeklyReorderPointPoolMonthlyRepository implements
    WeeklyReorderPointPoolMonthlyRepository {

  @Delegate
  private final MybatisWeeklyReorderPointPoolMonthlyMapper mapper;

  /**
   * PKを渡しての検索
   *
   * @param context        context
   * @param subsidiaryCode 現法コード
   * @param ginnerCode     統合インナーコード
   * @param plantCode      置場コード
   * @return 処理結果
   */
  @Override
  public Optional<WeeklyReorderPointPoolDataMonthlyEntity> findOne(RequestContext context,
                                                                   @NonNull String subsidiaryCode
      , @NonNull String ginnerCode
      , @NonNull String plantCode
  ) {
    return Optional.ofNullable(mapper.select(subsidiaryCode, ginnerCode, plantCode));
  }
}
