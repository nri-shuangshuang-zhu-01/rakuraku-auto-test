package jp.co.misumi.replenishstock.store.repository.mybatis;

import jp.co.misumi.fw.core.iterable.CloseableIterable;
import jp.co.misumi.fw.core.iterable.DelegatingCloseableIterable;
import jp.co.misumi.replenishstock.domain.entity.SearchConditionParamEntity;
import jp.co.misumi.replenishstock.domain.entity.WeeklyReorderPointExcludeConditionEntity;
import jp.co.misumi.replenishstock.domain.repository.CreateWeeklyOrderDlFileRepository;
import lombok.RequiredArgsConstructor;
import lombok.experimental.Delegate;

@RequiredArgsConstructor
public class MybatisCreateWeeklyOrderDlFileRepository implements CreateWeeklyOrderDlFileRepository {

  @Delegate
  private final MybatisCreateWeeklyOrderDlFileMapper mapper;

  /**
   * 出力ファイル情報を取得.
   *
   * @param searchConditionParamEntity 検索パラメター
   * @return 取得結果
   */
  @Override
  public CloseableIterable<WeeklyReorderPointExcludeConditionEntity> searchWeeklyReorderPoint(
      SearchConditionParamEntity searchConditionParamEntity) {

    return new DelegatingCloseableIterable<>(mapper.selectWeeklyReorderPoint(searchConditionParamEntity));
  }
}
