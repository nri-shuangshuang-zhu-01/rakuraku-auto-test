package jp.co.misumi.replenishstock.client;

import java.util.Map;
import javax.inject.Named;
import jp.co.misumi.fw.common.context.domain.RequestContext;
import jp.co.misumi.fw.rest.api.client.context.HttpHeaderMapCreator;
import jp.co.misumi.replenishstock.client.rest.product.internal.api.ProductSupplierInternalApi;
import jp.co.misumi.replenishstock.client.value.ClientEndpoint;
import jp.co.misumi.replenishstock.domain.client.ProductSupplierInternalClient;
import jp.co.misumi.replenishstock.model.rest.client.product.internal.GetListProductSupplierInternalRequestV1;
import jp.co.misumi.replenishstock.model.rest.client.product.internal.GetListProductSupplierInternalResponseV1;
import lombok.RequiredArgsConstructor;
import org.springframework.web.client.RestClientException;

@RequiredArgsConstructor
@Named
public class DefaultProductSupplierInternalClient implements ProductSupplierInternalClient {

  private final HttpHeaderMapCreator httpHeaderMapCreator;
  private final ProductSupplierInternalApi productSupplierInternalApi;

  @Override
  public GetListProductSupplierInternalResponseV1 getLProductSInternalV1(
      RequestContext context,
      GetListProductSupplierInternalRequestV1 getListProductSupplierInternalRequestV1)
      throws RestClientException {
    Map<String, Object> headerMap = httpHeaderMapCreator.createHeaderMap(context, null);
    // 外部APIの呼び出すー商品別仕入先情報取得API（List）
    return productSupplierInternalApi.getListProductSupplierInternalV1(
        context,
        ClientEndpoint.CLIENT_ENDPOINT_GET_LIST_PRODUCT.getValue(),
        headerMap,
        getListProductSupplierInternalRequestV1);
  }
}
