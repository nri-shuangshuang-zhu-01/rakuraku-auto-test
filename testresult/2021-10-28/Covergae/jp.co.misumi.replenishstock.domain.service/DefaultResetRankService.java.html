      RequestContext requestContext,
      String operationDate,
      List<ReorderPointEntity> reorderPointList,
      Map<String, ProductInternalV1> productInternalMap)
      throws BusinessException {

    int getCount = ReplenishStockConstants.INT_0; // 取得件数
    int updateCount = ReplenishStockConstants.INT_0; // 更新件数
    int skipCount = ReplenishStockConstants.INT_0; // SKIP件数
    int errCount = ReplenishStockConstants.INT_0; // エラー件数

    // 現法コード
    String subsidiaryCd;
    // 統合インナーコード
    String ginnerCode;
    // 置場コード
    String plantCd;
    // ランク
    String rank;
    // 発売年月日
    LocalDate releaseDate;
    // 発注点ロジックフラグ
    String reorderPointLogicFlag;
    // 製作品除外フラグ
    String excludeHalfStockFlag;

    List<UpdateStockInfoRequestDetailV1> detailList;
    UpdateStockInfoRequestV1 updateRequest;
    UpdateStockInfoRequestDetailV1 updateDetailV1;
    UpdateStockInfoResponseV1 updateStockInfoResponseV1;
    for (ReorderPointEntity pointEntity : reorderPointList) {
      ProductInternalV1 ginnerInternal = productInternalMap.get(pointEntity.getGinnerCode());
      if (ginnerInternal != null) {
        getCount += ReplenishStockConstants.INT_1;

        subsidiaryCd = pointEntity.getSubsidiaryCode();
        ginnerCode = pointEntity.getGinnerCode();
        plantCd = pointEntity.getPlantCode();
        rank = pointEntity.getRank();

        releaseDate = ginnerInternal.getReleaseDate();
        reorderPointLogicFlag = ginnerInternal.getReorderPointLogicFlag();
        excludeHalfStockFlag = ginnerInternal.getMtoExcludeFlag();

        String operationDateTemp = afterString(operationDate);

        DateTimeFormatter fmt =
            DateTimeFormatter.ofPattern(ReplenishStockConstants.PURE_DATE_PATTERN);
        LocalDate operationDatetime = LocalDate.parse(operationDateTemp, fmt);
        if (ReplenishStockConstants.STRING_2.equals(reorderPointLogicFlag)) {
          if (Period.between(releaseDate, operationDatetime).getMonths()
                  >= ReplenishStockConstants.INT_7
              && operationDatetime.isAfter(releaseDate)
              && !ReplenishStockConstants.RANK_E.equals(rank)) {
            String obtainSetRank =
                getSettingRank(
                    requestContext, subsidiaryCd, ginnerCode, plantCd, excludeHalfStockFlag);
            if (Objects.equals(obtainSetRank, pointEntity.getRank())) {
              skipCount += ReplenishStockConstants.INT_1;
            } else {
              updateRequest = new UpdateStockInfoRequestV1();
              updateDetailV1 = new UpdateStockInfoRequestDetailV1();
              detailList = new ArrayList<>();
              updateDetailV1.setRank(obtainSetRank);
              updateDetailV1.setPlantCode(plantCd);
              updateDetailV1.setGinnerCode(ginnerCode);
              detailList.add(updateDetailV1);
              updateRequest.setDetailList(detailList);
              try {
                updateStockInfoResponseV1 =
                    stockInfoClient.updateStockInfoV1(requestContext, updateRequest);
                updateCount++;
              } catch (Exception e) {
                errCount += ReplenishStockConstants.INT_1;
                messageLogger.log(messageSource, ReplenishStockMessages.BIR05000006E, null);
                throw new BusinessException(messageSource, ReplenishStockMessages.BIR05000006E, e);
              }
              if (CollectionUtils.isEmpty(updateStockInfoResponseV1.getMessageList())) {
                messageLogger.log(messageSource, ReplenishStockMessages.BIR05000006E, null);
                throw new BusinessException(
                    messageSource, ReplenishStockMessages.BIR05000006E, null);
              }
            }
          }
        }
      }
    }

    messageLogger.log(
        messageSource,
        ReplenishStockMessages.BIR05000005I,
        null,
        getCount,
        skipCount,
        updateCount,
        errCount);
  }
