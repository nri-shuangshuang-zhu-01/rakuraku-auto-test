  @Override
  public boolean calculation(
      RequestContext context, String operationDate, String subsidiaryCode, String jobNetId)
      throws BusinessException {

    log.info("週次発注点計算処理が開始しました。");

    // 処理対象データ取得
    final List<ReorderPointHandleDataEntity> handleDataList = getHandleData(context, operationDate);

    // 更新データ件数 = 在庫マスタの取得データ件数 - 未更新次数
    int saveCount = handleDataList.size();

    // 取得件数 = 在庫マスタの取得データ件数(在庫参照APIの取得件数)
    int getCount = handleDataList.size();

    if (CollectionUtils.isEmpty(handleDataList)) {
      outLog(getCount, saveCount);
      return false;
    }

    // 処理対象を繰り返す
    for (ReorderPointHandleDataEntity handleDataEntity : handleDataList) {

      // マスタ検索処理
      // 検索パラメータ.ランク　＝　"Z"　の場合
      if (ReplenishStockConstants.SEARCH_PARAM_RANK.equals(
          handleDataEntity.getReorderPointRank())) {

        // Zランク発注点計算処理」を行う
        if (Boolean.FALSE.equals(handleFlagReorderPointCalculation(handleDataEntity, context))) {

          // 在庫情報更新API異常の場合、更新次数マイナス1
          saveCount--;
        }

      } else {

        // マスタ検索処理、検索パラメータ.ランク != "Z"　の場合.
        if (!searchMasterHandleRankNormal(handleDataEntity, context)) {

          // 対象安全係数において、取得された汎用マスタ安全係数の検索結果が数値でない場合、対象データの処理を終了する。
          // 更新次数マイナス1
          saveCount--;
          continue;
        }

        // 平均出荷数、標準偏差算出処理
        if (Boolean.FALSE.equals(
            calculateAvgShipmentsAndStandardDeviation(operationDate, handleDataEntity, context))) {

          // 在庫情報更新API異常の場合、更新次数マイナス1
          saveCount--;
        }
      }
    }

    // 終了処理
    outLog(getCount, saveCount);

    return true;
  }

  /**
   * 終了ログ、処理件数、更新件数を出力する。.
   *
   * @param getCount 在庫マスタの取得データ件数
   * @param saveCount 更新データ件数
   */
  private void outLog(int getCount, int saveCount) {

    // 終了メッセージ：「週次発注点計算処理が終了しました。
    messageLogger.log(messageSource, ReplenishStockMessages.BIR17000011I, null);

    // 「取得件数：{0}件、更新件数:{1}件」、{0} : 在庫マスタの取得データ件数、{1} : 更新データ件数
    messageLogger.log(
        messageSource, ReplenishStockMessages.BIR17000012I, null, getCount, saveCount);
  }

  /**
   * 平均出荷数、標準偏差の算出処理.
   *
   * @param operationDate 運用日付
   * @param handleDataEntity 処理データ
   * @param context 共通
   */
  private Boolean calculateAvgShipmentsAndStandardDeviation(
      String operationDate, ReorderPointHandleDataEntity handleDataEntity, RequestContext context) {
    // 1.※対象期間(週数)の算出
    BigDecimal weekNumber = null;

    // 商品マスタ．発売年月日
    final LocalDate releaseDate = handleDataEntity.getReleaseDate();

    // 運用日
    final LocalDate operationDateLocalDate =
        LocalDate.parse(operationDate, DateTimeFormatter.ofPattern(DateUtil.SHORT_DATE_FORMAT));

    // 週数(小数第1位で切り上げ)　＝　(　運用日　－　商品マスタ．発売年月日　)　/　7
    weekNumber =
        NumberUtil.toBigDecimal(
            ((double) operationDateLocalDate.toEpochDay() - releaseDate.toEpochDay()) / 7);
    final BigDecimal weekNumberAdjust = NumberUtil.adjust(weekNumber, 0);

    // 週数　＜　52　の場合、内部変数．対象期間 = 週数
    if (NumberUtil.lt(weekNumber, new BigDecimal(52))) {
      handleDataEntity.setTargetPeriodVar(weekNumberAdjust);
    } else {
      // 上記以外の場合、内部変数．対象期間 = 52
      handleDataEntity.setTargetPeriodVar(new BigDecimal(52));
    }

    // 「週次発注点用POOLデータ(週次)」テーブルを検索する。
    final Optional<WeeklyReorderPointPoolDataWeeklyEntity> reorderPointDataWeeklyEntity =
        reorderPointWeeklyRepository.findOne(
            context,
            handleDataEntity.getReorderPointSubsidiaryCode(),
            handleDataEntity.getReorderPointGinnerCode(),
            handleDataEntity.getReorderPointPlantCode());

    // 週次発注点用POOLデータ(週次)判断用エンテイテイ
    WeeklyReorderPointPoolDataWeeklyEntity judgeEntity = null;

    ReorderPointWeeklyMonthlyEntity reorderPointWeeklyMonthlyEntity = null;
    if (reorderPointDataWeeklyEntity.isPresent()) {
      judgeEntity = reorderPointDataWeeklyEntity.get();

      reorderPointWeeklyMonthlyEntity =
          weeklyMonthlyEntityMapping.databaseToReorderPointEntity(judgeEntity, null);
    }

    // 「週次発注点用POOLデータ(週次)」テーブルに検索結果が存在の場合
    if (judgeEntity != null) {

      // 2.※対象期間内出荷数の算出
      handleDataEntity.setShipmentNumberVar(
          getShipmentNumberVar(reorderPointWeeklyMonthlyEntity, handleDataEntity));

      // 3.※平均出荷数の算出
      // 内部変数．平均出荷数(小数第5位で四捨五入)　＝　内部変数．対象期間内出荷数　/　内部変数．対象期間
      BigDecimal avgShipment =
          NumberUtil.divide(
              handleDataEntity.getShipmentNumberVar(), handleDataEntity.getTargetPeriodVar(), 4);
      handleDataEntity.setAvgShipmentVar(avgShipment);

      // 4.※分散集計の算出
      handleDataEntity.setDistributedAggregationVar(
          getDistributedAggregation(reorderPointWeeklyMonthlyEntity, handleDataEntity));

      // 5.※標本分散の算出
      // 内部変数．標本分散(小数第5位で四捨五入)　＝　内部変数．分散集計　/　(　内部変数．対象期間　－　1　)
      final BigDecimal sampleVariance =
          NumberUtil.divide(
              handleDataEntity.getDistributedAggregationVar(),
              (handleDataEntity.getTargetPeriodVar().subtract(BigDecimal.ONE)),
              4);
      handleDataEntity.setSampleVarianceVar(sampleVariance);

      // 6.※標準偏差の算出
      // 内部変数．標準偏差(小数第6位で四捨五入)　＝　√内部変数．標本分散
      final BigDecimal standardDeviation =
          NumberUtil.adjust(NumberUtil.sqrt(handleDataEntity.getSampleVarianceVar()), 5);
      handleDataEntity.setStandardDeviationVar(standardDeviation);

      // 発注点計算処理(発注点)
      if (Boolean.FALSE.equals(handleReorderPointCalculation(handleDataEntity, context))) {
        return false;
      }
    } else {

      // 出庫累積データ不在データ処理」を行う
      // 内部変数.発注点 = 0　とする
      handleDataEntity.setReorderPointVar(ReplenishStockConstants.REORDER_POINT_ZERO);

      // 発注点更新処理を行う
      if (Boolean.FALSE.equals(saveStockInfo(context, handleDataEntity))) {
        return false;
      }
    }
    return true;
  }

  /**
   * 分散集計の算出.
   *
   * @param reorderPointWeeklyMonthlyEntity 週次発注点用POOLデータ（週次）(月次)
   * @return 取得結果
   */
  private BigDecimal getDistributedAggregation(
      ReorderPointWeeklyMonthlyEntity reorderPointWeeklyMonthlyEntity,
      ReorderPointHandleDataEntity handleDataEntity) {
    // 内部変数.分散集計　＝　(　週次発注点計算用POOLデータ(週次)．在庫品出荷数(1)　－　内部変数.平均出荷数　)^2　～　(
    // 週次発注点計算用POOLデータ(週次)．在庫品出荷数(T)　－　内部変数平均出荷数　)^2　の集計値

    double distributedAggregation = 0;

    // 内部変数．対象期間をTとして計算

    // 在庫出荷数リスト
    final List<BigDecimal> stockShipmentQtyList =
        reorderPointWeeklyMonthlyEntity.getStockShipmentQtyList();

    // 製作品扱い出荷数リスト
    final List<BigDecimal> mtoModeShipmentQtyList =
        reorderPointWeeklyMonthlyEntity.getMtoModeShipmentQtyList();

    // 内部変数.対象期間をTとして計算
    for (int i = 0;
        i < NumberUtil.toInteger(handleDataEntity.getTargetPeriodVar().toString());
        i++) {

      // 内部変数．製作品扱除外フラグ＝"1"の場合
      if (ReplenishStockConstants.PRODUCT_EXCLUDE_FLAG_ONE.equals(
          handleDataEntity.getProductExcludeFlag())) {

        // 内部変数.分散集計　＝　(　週次発注点計算用POOLデータ(週次)．在庫品出荷数(1)　－　内部変数.平均出荷数　)^2　～　(
        // 週次発注点計算用POOLデータ(週次)．在庫品出荷数(T)　－　内部変数平均出荷数　)^2　の集計値
        distributedAggregation +=
            Math.pow(
                NumberUtil.toDouble(
                    stockShipmentQtyList.get(i).subtract(handleDataEntity.getAvgShipmentVar())),
                2);
      } else {
        // 上記以外の場合
        // 内部変数．分散集計　＝　(　週次発注点計算用POOLデータ(週次)．在庫品出荷数(1)　＋　週次発注点計算用POOLデータ(週次)．製作品扱出荷数(1)　－
        // 　内部変数．平均出荷数
        // 　)^2　～　(　週次発注点計算用POOLデータ(週次)．在庫品出荷数(T)　＋　週次発注点計算用POOLデータ(週次)．製作品扱出荷数(T)　－　内部変数.平均出荷数
        // 　)^2
        // 　の集計値

        distributedAggregation +=
            Math.pow(
                NumberUtil.toDouble(
                    stockShipmentQtyList
                        .get(i)
                        .add(mtoModeShipmentQtyList.get(i))
                        .subtract(handleDataEntity.getAvgShipmentVar())),
                2);
      }
    }
    return NumberUtil.toBigDecimal(distributedAggregation);
  }

  /**
   * 発注点計算処理(発注点).
   *
   * @param handleDataEntity 処理結果
   */
  private Boolean handleReorderPointCalculation(
      ReorderPointHandleDataEntity handleDataEntity, RequestContext context) {
    // 1.安全在庫の算出
    // 内部変数．安全在庫(小数第3位で四捨五入)　＝　内部変数．標準偏差　＊　内部係数．安全係数　＊　√(　内部変数．調達期間1　/　5　)
    // 平方根計算は小数第6位で四捨五入する。
    final BigDecimal safetyInventory =
        NumberUtil.adjust(
            handleDataEntity
                .getStandardDeviationVar()
                .multiply(handleDataEntity.getSafetyCoefficientVar())
                .multiply(
                    NumberUtil.adjust(
                        NumberUtil.sqrt(
                            handleDataEntity.getProcurementPeriod().divide(new BigDecimal(5))),
                        5)),
            2);
    handleDataEntity.setSafetyInventoryVar(safetyInventory);

    // 2.調達期間中出荷数1の算出、内部変数．調達期間中出荷数(小数第3位で四捨五入)　＝　内部変数．需要予測数量　＊　内部変数．調達期間　/　22
    BigDecimal duringShipmentPeriod =
        NumberUtil.divide(
            handleDataEntity
                .getDemandForecastQtyVar()
                .multiply(handleDataEntity.getProcurementPeriod()),
            new BigDecimal(22),
            2);

    handleDataEntity.setDuringShipmentPeriodVar(duringShipmentPeriod);

    // 3.発注点の算出: 内部変数.発注点(整数に切り上げ)　＝　内部変数.調達期間中出荷数　＋　内部変数.安全在庫
    final double reorderPointVarDouble =
        Math.ceil(
            NumberUtil.toDouble(
                handleDataEntity
                    .getDuringShipmentPeriodVar()
                    .add(handleDataEntity.getSafetyInventoryVar())));
    handleDataEntity.setReorderPointVar((int) reorderPointVarDouble);

    // 発注点更新処理を行う。
    return saveStockInfo(context, handleDataEntity);
  }

  /**
   * 対象期間内出荷数の算出.
   *
   * @param entity 週次発注点用POOLデータ(週次)(月次)
   * @param handleDataEntity 処理データ
   * @return 算出結果
   */
  private BigDecimal getShipmentNumberVar(
      ReorderPointWeeklyMonthlyEntity entity, ReorderPointHandleDataEntity handleDataEntity) {
    BigDecimal shipmentNumberVar = BigDecimal.ZERO;

    // 在庫出荷数リスト
    final List<BigDecimal> stockShipmentQtyList = entity.getStockShipmentQtyList();

    // 製作品扱い出荷数リスト
    final List<BigDecimal> mtoModeShipmentQtyList = entity.getMtoModeShipmentQtyList();

    // 内部変数．対象期間をTとして計算
    for (int i = 0;
        i < NumberUtil.toInteger(handleDataEntity.getTargetPeriodVar().toString());
        i++) {

      // 「週次発注点用POOLデータ(週次)」テーブルに検索結果が存在し、内部変数．製作品扱除外フラグ＝"0"の場合
      if (ReplenishStockConstants.PRODUCT_EXCLUDE_FLAG_ZERO.equals(
          handleDataEntity.getProductExcludeFlag())) {

        // 内部変数．対象期間内出荷数　＝　(　週次発注点計算用POOLデータ(週次).在庫品出荷数(1)　＋　週次発注点計算用POOLデータ(週次).製作品扱出荷数(1)　)　～　(
        // 週次発注点計算用POOLデータ(週次).在庫品出荷数(T)　＋　週次発注点計算用POOLデータ(週次).製作品扱出荷数(T)　)　の集計値
        shipmentNumberVar =
            shipmentNumberVar.add(stockShipmentQtyList.get(i).add(mtoModeShipmentQtyList.get(i)));
      } else {

        // 上記以外の場合
        // 内部変数．対象期間内出荷数　＝　週次発注点計算用POOLデータ(週次)．在庫品出荷数(1)　～　週次発注点計算用POOLデータ(週次)．在庫品出荷数(T)　の集計値
        shipmentNumberVar = shipmentNumberVar.add(stockShipmentQtyList.get(i));
      }
    }
    return shipmentNumberVar;
  }

  /**
   * マスタ検索処理、検索パラメータ.ランク != "Z"　の場合.
   *
   * @param handleDataEntity 処理対象データ
   * @param context 共通
   */
  private boolean searchMasterHandleRankNormal(
      ReorderPointHandleDataEntity handleDataEntity, RequestContext context) {

    // 安全係数テーブルから対象安全係数を取得する。
    // 検索条件.現法コード = 発注点.現法コード、商品所管部門コード = 商品マスタ.商品所管部門コード
    final Optional<SafetyCoefficientEntity> safetyCoefficientEntity =
        safetyCoefficientRepository.findOne(
            context,
            handleDataEntity.getReorderPointSubsidiaryCode(),
            handleDataEntity.getProductControlDepartmentCode());

    // 対象安全係数に存在する場合
    if (safetyCoefficientEntity.isPresent()) {

      if (Boolean.FALSE.equals(
          setSafetyCoefficient(safetyCoefficientEntity.get(), handleDataEntity))) {
        return false;
      }
    } else {
      // 対象安全係数に存在しない場合、内部変数．安全係数 = 3
      handleDataEntity.setSafetyCoefficientVar(ReplenishStockConstants.SAFETY_COEFFICIENT);
    }

    // 週次発注点用需要予測データを取得する
    final Optional<WeeklyReorderPointDemandForecastEntity> reorderPointForecastEntity =
        reorderPointForecastRepository.findOne(
            context,
            handleDataEntity.getReorderPointSubsidiaryCode(),
            handleDataEntity.getReorderPointGinnerCode(),
            handleDataEntity.getReorderPointPlantCode());

    // 週次発注点用需要予測データに検索結果が存在の場合: 内部変数.需要予測数量 = 週次発注点用需要予測.需要予測数量
    if (reorderPointForecastEntity.isPresent()) {

      // 対象データの処理を終了する。
      handleDataEntity.setDemandForecastQtyVar(
          reorderPointForecastEntity.get().getDemandForecastQty());
    } else {

      // 週次発注点用需要予測データに検索結果が存在しない場合、内部変数.需要予測数量 = 0
      handleDataEntity.setDemandForecastQtyVar(ReplenishStockConstants.DEMAND_FORECAST_QTY);
    }

    return true;
  }

  /**
   * 次発注点用需要予測データに検索結果が存在の場合、このメソッドを呼び出し、内部変数.安全係数を設定する.
   *
   * @param safetyCoefficientEntity 安全係数エンテイテイ
   * @param handleDataEntity 処理対象データ
   * @return 数値外チェック結果
   */
  private Boolean setSafetyCoefficient(
      SafetyCoefficientEntity safetyCoefficientEntity,
      ReorderPointHandleDataEntity handleDataEntity) {

    // 安全係数１
    final BigDecimal safetyCoefficient1 = safetyCoefficientEntity.getSafetyCoefficient1();

    // 安全係数2
    final BigDecimal safetyCoefficient2 = safetyCoefficientEntity.getSafetyCoefficient2();

    // 「安全係数１」、「安全係数２」がNULLのレコードは存在できますので、NULLチェック（数値外チェック）
    if ((safetyCoefficient1 == null
            && ReplenishStockConstants.RANK_X.equals(handleDataEntity.getReorderPointRank()))
        || (safetyCoefficient2 == null
            && ReplenishStockConstants.RANK_Y.equals(handleDataEntity.getReorderPointRank()))) {

      // 取得された汎用マスタ安全係数の検索結果が数値でない場合、警告メッセージをログ出力する。
      // a) 警告メッセージをログ出力する。
      messageLogger.log(
          messageSource,
          ReplenishStockMessages.BIR17000006W,
          null,
          handleDataEntity.getReorderPointSubsidiaryCode(),
          handleDataEntity.getProductControlDepartmentCode());

      // 対象データの処理を終了する。
      return false;
    } else {
      // 検索パラメータ．ランク　＝　"X"　の場合、内部変数．安全係数 = 安全係数１
      if (ReplenishStockConstants.RANK_X.equals(handleDataEntity.getReorderPointRank())) {
        handleDataEntity.setSafetyCoefficientVar(safetyCoefficient1);
      }

      // 検索パラメータ．ランク　＝　"Y"　の場合、内部変数．安全係数 = 安全係数2
      if (ReplenishStockConstants.RANK_Y.equals(handleDataEntity.getReorderPointRank())) {
        handleDataEntity.setSafetyCoefficientVar(safetyCoefficient2);
      }
    }

    return true;
  }

  /**
   * Zランク発注点計算処理」を行う.
   *
   * @param handleDataEntity 処理データエンテイテイ
   */
  private Boolean handleFlagReorderPointCalculation(
      ReorderPointHandleDataEntity handleDataEntity, RequestContext context) {

    // 「週次発注点用POOLデータ(月次)」テーブルを検索する。
    final Optional<WeeklyReorderPointPoolDataMonthlyEntity> reorderPointMonthlyEntity =
        reorderPointMonthlyRepository.findOne(
            context,
            handleDataEntity.getReorderPointSubsidiaryCode(),
            handleDataEntity.getReorderPointGinnerCode(),
            handleDataEntity.getReorderPointPlantCode());

    // 週次発注点計算用POOLデータ(月次)」判断用エンテイテイ
    WeeklyReorderPointPoolDataMonthlyEntity judgeEntity = null;

    if (reorderPointMonthlyEntity.isPresent()) {
      judgeEntity = reorderPointMonthlyEntity.get();
    }

    // 「週次発注点計算用POOLデータ(月次)」テーブルに検索結果が存在し
    if (judgeEntity != null) {

      final ReorderPointWeeklyMonthlyEntity reorderPointWeeklyMonthlyEntity =
          weeklyMonthlyEntityMapping.databaseToReorderPointEntity(null, judgeEntity);

      handleDataEntity.setMaximumMonthlyShipmentVar(
          getMaximumMonthlyShipment(
              reorderPointWeeklyMonthlyEntity, handleDataEntity.getProductExcludeFlag()));
    } else {
      // 「週次発注点計算用POOLデータ(月次)」テーブルに検索結果が存在しない場合、内部変数.最大月間出荷数 = 0、外部qa:No.329
      handleDataEntity.setMaximumMonthlyShipmentVar(
          ReplenishStockConstants.MAXIMUM_MONTHLY_SHIPMENT);
    }

    // 発注点設定処理: 内部変数.発注点 = 内部変.最大月間出荷数とする。
    handleDataEntity.setReorderPointVar(
        NumberUtil.toInteger(handleDataEntity.getMaximumMonthlyShipmentVar().toString()));

    // 発注点更新処理
    return saveStockInfo(context, handleDataEntity);
  }

  /**
   * 最大月間出荷数を取得するメソッド.
   *
   * @param reorderPointWeeklyMonthlyEntity 週次発注点用POOLデータ(月次)(週次)
   * @return 取得結果
   */
  private BigDecimal getMaximumMonthlyShipment(
      ReorderPointWeeklyMonthlyEntity reorderPointWeeklyMonthlyEntity, String productExcludeFlag) {

    List<BigDecimal> maximumMonthlyShipmentList = new ArrayList<>();

    // 在庫出荷数リスト
    final List<BigDecimal> stockShipmentQtyList =
        reorderPointWeeklyMonthlyEntity.getStockShipmentQtyList();

    // 製作品扱い出荷数リスト
    final List<BigDecimal> mtoModeShipmentQtyList =
        reorderPointWeeklyMonthlyEntity.getMtoModeShipmentQtyList();

    for (int i = 0; i < 7; i++) {
      if (ReplenishStockConstants.PRODUCT_EXCLUDE_FLAG_ZERO.equals(productExcludeFlag)) {
        // 「週次発注点計算用POOLデータ(月次)」テーブルに検索結果が存在し、内部変数.製作品扱除外フラグ＝"0"の場合
        // 内部変数．最大月間出荷数　＝　(　週次発注点計算用POOLデータ(月次)．在庫品出荷数(1)　＋
        // 週次発注点計算用POOLデータ(月次)．製作品扱出荷数(1)　)～　(　週次発注点計算用POOLデータ(月次)．
        // 在庫品出荷数(7)　＋　週次発注点計算用POOLデータ(月次)．製作品扱出荷数(7)　)　の最大値

        maximumMonthlyShipmentList.add(
            stockShipmentQtyList.get(i).add(mtoModeShipmentQtyList.get(i)));
      } else {

        // 「週次発注点計算用POOLデータ(月次)」テーブルに検索結果が存在し、内部変数.製作品扱除外フラグ＝"0"以外の場合
        // 内部変数.最大月間出荷数　＝　週次発注点計算用POOLデータ(月次).在庫品出荷数(1)　～　週次発注点計算用POOLデータ(月次).在庫品出荷数(7)の最大値
        maximumMonthlyShipmentList.add(stockShipmentQtyList.get(i));
      }
    }
    Collections.sort(maximumMonthlyShipmentList);

    return maximumMonthlyShipmentList.get(maximumMonthlyShipmentList.size() - 1);
  }

        getSupplyPeriodMap(stockInfoMap, productSupplierMap, productMap, context);

        stockInfoMap.values();

      final ReorderPointHandleDataEntity reorderPointHandleDataEntity =
          handleDataEntityMapping.responseToReorderPointHandleDataEntity(
              productMap.get(ginnerCode),
              productSupplierMap.get(ginnerCode),
              stockInfoMap.get(ginnerCode),

      reorderPointHandleDataEntity.setReorderPointSubsidiaryCode(context.getSubsidiaryCode());

    }
